    Disruption Event System¶

    from enum import Enum, auto
from dataclasses import dataclass
from typing import List, Dict, Optional, Callable
import numpy as np

class DisruptionType(Enum):
    """Types of disruptions that can occur."""
    VEHICLE_BREAKDOWN = auto()
    CHARGING_STATION_FAILURE = auto()
    ROAD_CLOSURE = auto()
    DEMAND_SPIKE = auto()
    DRIVER_UNAVAILABLE = auto()
    POWER_OUTAGE = auto()
    WEATHER_SEVERE = auto()

@dataclass
class DisruptionEvent:
    """Represents a disruption in the network."""
    event_id: str
    disruption_type: DisruptionType
    start_time: float
    duration_min: float
    severity: float  # 0.0 to 1.0
    affected_entities: List[str]  # IDs of affected vehicles/stations/roads
    location: Optional[Tuple[float, float]] = None
    radius_km: Optional[float] = None  # For area effects

    Resilience Manager¶

    class ResilienceManager:
    """
    Manages disruption events and measures system resilience.
    """

    def __init__(self):
        self.active_disruptions: List[DisruptionEvent] = []
        self.disruption_history: List[DisruptionEvent] = []
        self.baseline_metrics: Dict = {}
        self.stress_metrics: Dict = {}

    def inject_disruption(self, disruption: DisruptionEvent):
        """Inject a disruption into the simulation."""
        self.active_disruptions.append(disruption)

        # Apply disruption effects
        if disruption.disruption_type == DisruptionType.CHARGING_STATION_FAILURE:
            self._apply_charging_failure(disruption)
        elif disruption.disruption_type == DisruptionType.VEHICLE_BREAKDOWN:
            self._apply_vehicle_breakdown(disruption)
        elif disruption.disruption_type == DisruptionType.DEMAND_SPIKE:
            self._apply_demand_spike(disruption)

        self.logger.warning(
            f"Disruption injected: {disruption.disruption_type.name} "
            f"affecting {len(disruption.affected_entities)} entities",
            tag="resilience"
        )

    def _apply_charging_failure(self, disruption: DisruptionEvent):
        """Apply charging station failure effects."""
        for station_id in disruption.affected_entities:
            station = self.infrastructure_controller.get_station(station_id)
            if station:
                # Reduce available chargers based on severity
                original_chargers = station.num_chargers
                failed_chargers = int(original_chargers * disruption.severity)
                station.chargers_available = max(0, original_chargers - failed_chargers)
                station.state = ChargingStationState.DEGRADED

    def _apply_vehicle_breakdown(self, disruption: DisruptionEvent):
        """Apply vehicle breakdown effects."""
        for vehicle_id in disruption.affected_entities:
            vehicle = self.fleet_controller.get_vehicle(vehicle_id)
            if vehicle:
                vehicle.state = VehicleState.BROKEN_DOWN
                # Trigger reassignment of vehicle's demands
                self._reassign_vehicle_demands(vehicle)

    def measure_resilience(self) -> Dict:
        """
        Calculate resilience metrics comparing stressed vs baseline performance.
        """
        current_metrics = self._calculate_current_metrics()

        resilience_scores = {
            'service_level_retention': (
                current_metrics['on_time_rate'] /
                self.baseline_metrics.get('on_time_rate', 1.0)
            ),
            'capacity_retention': (
                current_metrics['completed_deliveries'] /
                self.baseline_metrics.get('completed_deliveries', 1)
            ),
            'cost_increase_factor': (
                current_metrics['total_cost'] /
                self.baseline_metrics.get('total_cost', 1)
            ),
            'recovery_time_estimate': self._estimate_recovery_time(),
            'alternative_routes_used': current_metrics['reroute_count'],
            'cascade_failures': self._count_cascade_failures()
        }

        return resilience_scores

    Fleet Strategy Comparator¶

    @dataclass
class FleetStrategy:
    """Defines a fleet composition and operational strategy."""
    strategy_id: str
    name: str
    fleet_composition: Dict[str, int]  # vehicle_spec_id -> count
    charging_strategy: str  # "depot_only", "opportunistic", "scheduled"
    operational_hours: Tuple[int, int]  # (start_hour, end_hour)
    infrastructure_level: str  # "minimal", "moderate", "comprehensive"

class FleetStrategyComparator:
    """
    Compares different fleet strategies across multiple scenarios.
    """

    def __init__(self):
        self.strategies: List[FleetStrategy] = []
        self.scenario_results: Dict[str, Dict] = {}

    def add_strategy(self, strategy: FleetStrategy):
        """Add a fleet strategy for comparison."""
        self.strategies.append(strategy)

    def run_comparison(
        self,
        demand_scenarios: List[DemandScenario],
        disruption_scenarios: List[DisruptionScenario],
        simulation_days: int = 30
    ) -> Dict:
        """
        Run all strategies through all scenarios and compare.
        """
        results = {}

        for strategy in self.strategies:
            strategy_results = {
                'baseline': {},
                'disrupted': {},
                'averages': {}
            }

            # Run baseline scenarios
            for demand_scenario in demand_scenarios:
                scenario_key = f"demand_{demand_scenario.name}"
                metrics = self._run_strategy_scenario(
                    strategy, demand_scenario, None, simulation_days
                )
                strategy_results['baseline'][scenario_key] = metrics

            # Run disruption scenarios
            for disruption_scenario in disruption_scenarios:
                scenario_key = f"disruption_{disruption_scenario.name}"
                metrics = self._run_strategy_scenario(
                    strategy, demand_scenarios[0], disruption_scenario, simulation_days
                )
                strategy_results['disrupted'][scenario_key] = metrics

            # Calculate averages and summaries
            strategy_results['averages'] = self._calculate_strategy_averages(
                strategy_results
            )

            results[strategy.strategy_id] = strategy_results

        # Generate comparison report
        return self._generate_comparison_report(results)

    def _run_strategy_scenario(
        self,
        strategy: FleetStrategy,
        demand_scenario: DemandScenario,
        disruption_scenario: Optional[DisruptionScenario],
        days: int
    ) -> Dict:
        """Run a single strategy through a scenario."""

        # Initialize simulation with strategy
        sim = self._create_simulation(strategy)

        # Apply demand scenario
        sim.demand_controller.set_demand_pattern(demand_scenario)

        # Apply disruptions if specified
        if disruption_scenario:
            sim.resilience_manager.schedule_disruptions(disruption_scenario)

        # Run simulation
        sim.run(days=days)

        # Collect metrics
        return {
            'total_deliveries': sim.get_total_deliveries(),
            'on_time_rate': sim.get_on_time_rate(),
            'total_cost': sim.get_total_cost(),
            'cost_per_delivery': sim.get_cost_per_delivery(),
            'co2_emissions': sim.get_total_emissions(),
            'vehicle_utilization': sim.get_average_utilization(),
            'energy_consumption': sim.get_total_energy(),
            'failed_deliveries': sim.get_failed_deliveries(),
            'recovery_events': sim.get_recovery_count()
        }

    Cost Analysis System¶

    class CostAnalyzer:
    """
    Comprehensive cost tracking and analysis.
    """

    def __init__(self):
        self.cost_categories = {
            'capital': {},
            'operational': {},
            'penalty': {},
            'opportunity': {}
        }

    def track_vehicle_costs(self, vehicle: VehicleView, dt: float):
        """Track all costs associated with a vehicle."""

        vehicle_id = vehicle.vehicle_id
        spec = self.vehicle_catalog.get_spec(vehicle.spec_id)

        # Capital cost amortization (per hour)
        if vehicle_id not in self.cost_categories['capital']:
            # Initialize capital cost tracking
            if spec.lease_price_monthly_gbp > 0:
                hourly_lease = spec.lease_price_monthly_gbp / (30 * 24)
            else:
                # Assume 5-year depreciation
                hourly_depreciation = spec.purchase_price_gbp / (5 * 365 * 24)
                hourly_lease = hourly_depreciation

            self.cost_categories['capital'][vehicle_id] = 0

        self.cost_categories['capital'][vehicle_id] += hourly_lease * (dt / 3600)

        # Operational costs
        if vehicle.state == VehicleState.EN_ROUTE:
            # Energy costs
            energy_cost = self._calculate_energy_cost(vehicle, dt)
            self._add_operational_cost(vehicle_id, 'energy', energy_cost)

            # Driver costs
            driver_cost = 25.0 * (dt / 3600)  # £25/hour
            self._add_operational_cost(vehicle_id, 'driver', driver_cost)

            # Wear and maintenance
            distance_km = vehicle.speed_kmh * (dt / 3600)
            maintenance_cost = distance_km * 0.05  # £0.05/km
            self._add_operational_cost(vehicle_id, 'maintenance', maintenance_cost)

    def _calculate_energy_cost(self, vehicle: VehicleView, dt: float) -> float:
        """Calculate energy cost based on vehicle type."""

        distance_km = vehicle.speed_kmh * (dt / 3600)

        if vehicle.propulsion_type == "BEV":
            # Electricity cost
            kwh_consumed = distance_km * 1.2  # kWh/km
            if vehicle.state == VehicleState.CHARGING:
                # Rapid charging premium
                cost_per_kwh = 0.45
            else:
                # Depot charging
                cost_per_kwh = 0.25
            return kwh_consumed * cost_per_kwh

        elif vehicle.propulsion_type == "HFCEV":
            # Hydrogen cost
            kg_consumed = distance_km * 0.08  # kg/km
            cost_per_kg = 10.0  # £10/kg
            return kg_consumed * cost_per_kg

        elif vehicle.propulsion_type == "DIESEL":
            # Diesel cost
            liters_consumed = distance_km * 0.35  # L/km
            cost_per_liter = 1.50
            return liters_consumed * cost_per_liter

        return 0

    def calculate_tco(self, vehicle_id: str, years: int = 5) -> Dict:
        """Calculate Total Cost of Ownership for a vehicle."""

        vehicle = self.fleet_controller.get_vehicle(vehicle_id)
        spec = self.vehicle_catalog.get_spec(vehicle.spec_id)

        # Capital costs
        if spec.lease_price_monthly_gbp > 0:
            capital_cost = spec.lease_price_monthly_gbp * 12 * years
        else:
            capital_cost = spec.purchase_price_gbp
            # Add residual value
            residual = spec.purchase_price_gbp * (spec.residual_value_pct / 100)
            capital_cost -= residual

        # Operational costs (projected)
        annual_distance = vehicle.odometer_km * 365 / self.simulation_days

        # Energy
        if vehicle.propulsion_type == "BEV":
            annual_energy_cost = annual_distance * 1.2 * 0.25  # kWh/km * £/kWh
        elif vehicle.propulsion_type == "HFCEV":
            annual_energy_cost = annual_distance * 0.08 * 10.0  # kg/km * £/kg
        else:
            annual_energy_cost = annual_distance * 0.35 * 1.50  # L/km * £/L

        energy_total = annual_energy_cost * years

        # Maintenance
        maintenance_total = annual_distance * 0.05 * years

        # Driver costs (assuming 250 working days/year, 10 hours/day)
        driver_total = 25 * 10 * 250 * years

        return {
            'capital': capital_cost,
            'energy': energy_total,
            'maintenance': maintenance_total,
            'driver': driver_total,
            'total': capital_cost + energy_total + maintenance_total + driver_total,
            'per_km': (capital_cost + energy_total + maintenance_total) / (annual_distance * years),
            'per_year': (capital_cost + energy_total + maintenance_total + driver_total) / years
        }

    class ScenarioGenerator:
    """
    Generates test scenarios for resilience and strategy testing.
    """

    def generate_progressive_electrification(self) -> List[FleetStrategy]:
        """Generate strategies for progressive fleet electrification."""

        strategies = []
        base_fleet_size = 50

        for bev_percentage in [0, 20, 40, 60, 80, 100]:
            bev_count = int(base_fleet_size * bev_percentage / 100)
            diesel_count = base_fleet_size - bev_count

            strategy = FleetStrategy(
                strategy_id=f"electrification_{bev_percentage}",
                name=f"{bev_percentage}% Electric Fleet",
                fleet_composition={
                    'VOLVO_FH_BEV_540': bev_count // 2,
                    'DAIMLER_EACTROS_600': bev_count - (bev_count // 2),
                    'VOLVO_FH_DIESEL': diesel_count
                },
                charging_strategy="depot_only" if bev_percentage <= 40 else "opportunistic",
                operational_hours=(6, 22),
                infrastructure_level="minimal" if bev_percentage <= 20 else "moderate"
            )
            strategies.append(strategy)

        return strategies

    def generate_disruption_scenarios(self) -> List[DisruptionScenario]:
        """Generate standard disruption scenarios for testing."""

        scenarios = []

        # Infrastructure failure scenario

    Reporting and Visualization¶

    class ResilienceReport:
    """
    Generates comprehensive resilience analysis reports.
    """

    def generate_comparison_matrix(
        self,
        strategies: List[FleetStrategy],
        results: Dict
    ) -> pd.DataFrame:
        """Generate comparison matrix of strategies."""

        data = []
        for strategy in strategies:
            strategy_results = results[strategy.strategy_id]

            row = {
                'Strategy': strategy.name,
                'Fleet Size': sum(strategy.fleet_composition.values()),
                'BEV %': self._calculate_bev_percentage(strategy),
                'TCO (5yr)': strategy_results['averages']['tco_5yr'],
                'Cost/Delivery': strategy_results['averages']['cost_per_delivery'],
                'On-Time Rate': strategy_results['averages']['on_time_rate'],
                'CO2/tonne-km': strategy_results['averages']['co2_per_tonne_km'],
                'Resilience Score': strategy_results['averages']['resilience_score'],
                'Recovery Time': strategy_results['averages']['recovery_time_min']
            }
            data.append(row)

        df = pd.DataFrame(data)
        df = df.sort_values('TCO (5yr)')

        return df

    def generate_resilience_dashboard(self, results: Dict) -> Dict:
        """Generate dashboard data for resilience visualization."""

        dashboard = {
            'summary_metrics': {
                'best_tco_strategy': self._find_best_strategy(results, 'tco_5yr'),

                'lowest_emissions_strategy': self._find_best_strategy(results, 'co2_total'),
                'optimal_bev_percentage': self._find_optimal_bev_percentage(results)
            },
            'charts': {
                'tco_comparison': self._generate_tco_chart_data(results),

                'cost_breakdown': self._generate_cost_breakdown_data(results)

            'recommendations': self._generate_recommendations(results)
        }

        return dashboard
