    Vehicle Assignment Optimization¶

    Core Assignment Problem¶

    from dataclasses import dataclass
from typing import List, Dict, Optional, Tuple
import numpy as np

@dataclass
class AssignmentResult:
    """Result of vehicle assignment optimization."""
    vehicle_assignments: Dict[str, List[str]]  # vehicle_id -> demand_ids
    unassigned_demands: List[str]
    total_cost: float
    total_distance_km: float
    avg_utilization: float

    Vehicle Assignment Optimizer¶

    class VehicleAssignmentOptimizer:
    """
    Assigns demands to vehicles for a single operator.
    Optimizes for cost, feasibility, and service level.
    """

    def __init__(self):
        # UK operational parameters
        self.max_driving_time_min = 540  # 9 hours daily limit
        self.max_continuous_driving_min = 270  # 4.5 hours
        self.required_break_min = 45

    def assign_demands_to_vehicles(
        self,
        vehicles: List[VehicleView],
        demands: List[DemandView],
        infrastructure: Dict
    ) -> AssignmentResult:
        """
        Main assignment method using two-phase approach:
        1. Cluster demands geographically and temporally
        2. Assign best vehicle to each cluster
        """

        # Filter available vehicles
        available_vehicles = [v for v in vehicles if v.state == VehicleState.IDLE.value]

        if not available_vehicles:
            return AssignmentResult(
                vehicle_assignments={},
                unassigned_demands=[d.demand_id for d in demands],
                total_cost=0,
                total_distance_km=0,
                avg_utilization=0
            )

        # Phase 1: Cluster demands
        clusters = self._cluster_demands(demands)

        # Phase 2: Assign vehicles to clusters
        assignments = self._assign_vehicles_to_clusters(
            available_vehicles, clusters, infrastructure
        )

        # Calculate metrics
        metrics = self._calculate_assignment_metrics(assignments, demands)

        return AssignmentResult(
            vehicle_assignments=assignments,
            unassigned_demands=metrics['unassigned'],
            total_cost=metrics['total_cost'],
            total_distance_km=metrics['total_distance'],
            avg_utilization=metrics['avg_utilization']
        )

    Demand Clustering¶

    def _cluster_demands(self, demands: List[DemandView]) -> List[List[DemandView]]:
    """
    Cluster demands based on proximity and time windows.
    Uses DBSCAN for density-based clustering.
    """
    from sklearn.cluster import DBSCAN

    if len(demands) < 2:
        return [[d] for d in demands]

    # Build feature matrix: [x, y, time_window_center]
    features = []
    for demand in demands:
        x = demand.destination_x
        y = demand.destination_y
        # Normalize time to same scale as distance
        time_center = (demand.delivery_window_start + demand.delivery_window_end) / 2
        time_normalized = time_center / (24 * 3600) * 100  # Scale to ~100km

        features.append([x, y, time_normalized])

    features = np.array(features)

    # DBSCAN clustering
    clustering = DBSCAN(eps=30, min_samples=2).fit(features)
    labels = clustering.labels_

    # Group by cluster
    clusters = {}
    for demand, label in zip(demands, labels):
        if label == -1:  # Noise point - create individual cluster
            clusters[f"single_{demand.demand_id}"] = [demand]
        else:
            if label not in clusters:
                clusters[label] = []
            clusters[label].append(demand)

    return list(clusters.values())

    Vehicle-to-Cluster Assignment¶

    def _assign_vehicles_to_clusters(
    self,
    vehicles: List[VehicleView],
    clusters: List[List[DemandView]],
    infrastructure: Dict
) -> Dict[str, List[str]]:
    """
    Assign best vehicle to each demand cluster.
    Considers vehicle capabilities and constraints.
    """

    assignments = {}
    assigned_vehicles = set()

    # Sort clusters by priority and size
    sorted_clusters = sorted(
        clusters,
        key=lambda c: (
            -max(d.priority for d in c),  # Higher priority first
            -sum(d.weight_kg for d in c)   # Larger loads first
        )
    )

    for cluster in sorted_clusters:
        # Calculate cluster requirements
        cluster_weight = sum(d.weight_kg for d in cluster)
        cluster_volume = sum(d.volume_m3 for d in cluster)
        cluster_distance = self._estimate_cluster_distance(cluster)

        # Find best available vehicle
        best_vehicle = None
        best_score = float('inf')

        for vehicle in vehicles:
            if vehicle.vehicle_id in assigned_vehicles:
                continue

            # Check feasibility
            if not self._is_vehicle_feasible_for_cluster(
                vehicle, cluster, cluster_weight, cluster_distance
            ):
                continue

            # Calculate assignment score
            score = self._calculate_assignment_score(
                vehicle, cluster, cluster_distance, infrastructure
            )

            if score < best_score:
                best_score = score
                best_vehicle = vehicle

        # Make assignment
        if best_vehicle:
            assignments[best_vehicle.vehicle_id] = [d.demand_id for d in cluster]
            assigned_vehicles.add(best_vehicle.vehicle_id)

    return assignments

    Feasibility Checking¶

    def _is_vehicle_feasible_for_cluster(
    self,
    vehicle: VehicleView,
    cluster: List[DemandView],
    total_weight: float,
    estimated_distance: float
) -> bool:
    """
    Check if vehicle can feasibly serve the cluster.
    """

    # Get vehicle specification
    spec = self.vehicle_catalog.get_spec(vehicle.spec_id)

    # Check weight capacity
    if total_weight > spec.payload_capacity_kg:
        return False

    # Check range for BEVs
    if vehicle.propulsion_type == "BEV":
        energy_required = estimated_distance * spec.primary_propulsion.consumption_baseline
        battery_energy = spec.primary_propulsion.battery_capacity_kwh * (vehicle.current_soc_pct / 100)

        # Need 20% buffer
        if energy_required > battery_energy * 0.8:
            return False

    # Check range for HFCEVs
    elif vehicle.propulsion_type == "HFCEV":
        h2_required = estimated_distance * spec.primary_propulsion.consumption_baseline
        if h2_required > vehicle.current_h2_level_kg * 0.8:
            return False

    # Check time feasibility
    estimated_time = self._estimate_delivery_time(cluster, estimated_distance)
    if estimated_time > self.max_driving_time_min:
        return False

    # Check time windows
    earliest_start = min(d.delivery_window_start for d in cluster)
    latest_end = max(d.delivery_window_end for d in cluster)
    window_span = (latest_end - earliest_start) / 3600  # hours

    if window_span > 10:  # Too spread out
        return False

    return True

    Assignment Scoring¶

    def _calculate_assignment_score(
    self,
    vehicle: VehicleView,
    cluster: List[DemandView],
    distance: float,
    infrastructure: Dict
) -> float:
    """
    Calculate score for assigning vehicle to cluster.
    Lower score is better.
    """

    spec = self.vehicle_catalog.get_spec(vehicle.spec_id)
    score = 0.0

    # Distance cost
    distance_cost = distance * 2.0  # £2 per km base cost
    score += distance_cost

    # Energy cost
    if vehicle.propulsion_type == "BEV":
        energy_kwh = distance * spec.primary_propulsion.consumption_baseline
        energy_cost = energy_kwh * 0.25  # £0.25 per kWh
        score += energy_cost

        # Charging time penalty if needed
        if vehicle.current_soc_pct < 50:
            score += 50  # Penalty for low SOC

    elif vehicle.propulsion_type == "HFCEV":
        h2_kg = distance * spec.primary_propulsion.consumption_baseline
        h2_cost = h2_kg * 10.0  # £10 per kg H2
        score += h2_cost

    elif vehicle.propulsion_type == "DIESEL":
        fuel_l = distance * spec.primary_propulsion.consumption_baseline
        fuel_cost = fuel_l * 1.5  # £1.50 per liter
        score += fuel_cost

    # Utilization bonus (prefer using vehicles efficiently)
    cluster_weight = sum(d.weight_kg for d in cluster)
    utilization = cluster_weight / spec.payload_capacity_kg
    if utilization > 0.7:  # Good utilization
        score *= 0.9  # 10% bonus

    # Time window tightness penalty
    time_flexibility = self._calculate_time_flexibility(cluster)
    if time_flexibility < 2:  # Less than 2 hours flexibility
        score *= 1.2  # 20% penalty

    return score

    Helper Methods¶

    def _estimate_cluster_distance(self, cluster: List[DemandView]) -> float:
    """
    Estimate total distance to serve all demands in cluster.
    Uses TSP approximation.
    """

    if len(cluster) == 1:
        demand = cluster[0]
        # Distance from depot to demand and back
        return 2 * np.sqrt(demand.destination_x**2 + demand.destination_y**2)

    # Build distance matrix
    n = len(cluster)
    positions = [(d.destination_x, d.destination_y) for d in cluster]

    # Add depot at origin
    positions.insert(0, (0, 0))

    # Calculate pairwise distances
    total_distance = 0
    current_pos = 0  # Start at depot

    unvisited = set(range(1, n + 1))

    # Nearest neighbor heuristic
    while unvisited:
        nearest = min(unvisited, key=lambda i: self._euclidean_distance(
            positions[current_pos], positions[i]
        ))
        total_distance += self._euclidean_distance(
            positions[current_pos], positions[nearest]
        )
        current_pos = nearest
        unvisited.remove(nearest)

    # Return to depot
    total_distance += self._euclidean_distance(positions[current_pos], positions[0])

    return total_distance

def _euclidean_distance(self, pos1: Tuple[float, float], pos2: Tuple[float, float]) -> float:
    """Calculate Euclidean distance between two positions."""
    return np.sqrt((pos1[0] - pos2[0])**2 + (pos1[1] - pos2[1])**2)

def _estimate_delivery_time(self, cluster: List[DemandView], distance: float) -> float:
    """
    Estimate time needed to complete all deliveries.
    """

    # Driving time at average speed
    avg_speed_kmh = 50
    driving_time = (distance / avg_speed_kmh) * 60  # minutes

    # Add time for each delivery
    delivery_time = len(cluster) * 15  # 15 minutes per delivery

    # Add break time if needed
    total_time = driving_time + delivery_time
    if total_time > self.max_continuous_driving_min:
        total_time += self.required_break_min

    return total_time

def _calculate_time_flexibility(self, cluster: List[DemandView]) -> float:
    """
    Calculate average time window flexibility in hours.
    """

    flexibilities = []
    for demand in cluster:
        window_hours = (demand.delivery_window_end - demand.delivery_window_start) / 3600
        flexibilities.append(window_hours)

    return np.mean(flexibilities)

    Optimization Result Processing¶

    def _calculate_assignment_metrics(
    self,
    assignments: Dict[str, List[str]],
    all_demands: List[DemandView]
) -> Dict:
    """
    Calculate metrics for the assignment result.
    """

    assigned_demand_ids = set()
    total_distance = 0
    total_cost = 0
    utilizations = []

    for vehicle_id, demand_ids in assignments.items():
        assigned_demand_ids.update(demand_ids)

        # Get assigned demands
        assigned = [d for d in all_demands if d.demand_id in demand_ids]

        # Calculate distance
        distance = self._estimate_cluster_distance(assigned)
        total_distance += distance

        # Calculate cost (simplified)
        cost = distance * 2.0  # Base cost
        total_cost += cost

        # Calculate utilization
        vehicle = self.get_vehicle(vehicle_id)
        spec = self.vehicle_catalog.get_spec(vehicle.spec_id)
        weight = sum(d.weight_kg for d in assigned)
        utilization = weight / spec.payload_capacity_kg
        utilizations.append(utilization)

    # Find unassigned demands
    all_demand_ids = {d.demand_id for d in all_demands}
    unassigned = list(all_demand_ids - assigned_demand_ids)

    return {
        'unassigned': unassigned,
        'total_cost': total_cost,
        'total_distance': total_distance,
        'avg_utilization': np.mean(utilizations) if utilizations else 0
    }
